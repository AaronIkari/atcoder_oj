### Problem B - Resale

```python
N = int(raw_input())

V = map(int, raw_input().split())
C = map(int, raw_input().split())

X_Y = 0

for i in range(N):
    if V[i] - C[i] > 0:
        X_Y += V[i] - C[i]

print '{}'.format(X_Y)
```

### Problem C - GCD on Blackboard
```python
N = int(input())
A = list(map(int, raw_input().split()))

def gcd(m,n):
    return m if n == 0 else gcd(n, m%n)

L = [0] * (N+1)
R = [0] * (N+1)

# L: list
# boundary: L[0] = 0, L[N-1] = 0
# L[i]: previous (i-1)-th gcd value, i = [0, ..., N-2]
for i in range(N-1):
    L[i+1] = gcd(L[i], A[i])

# R: list
# boundary: R[0] = 0, R[N-1] = 0
# R[i]: last (N-i)-th gcd value, i = [N-1, ..., 1]
for i in range(N-1, 0, -1):
    R[i] = gcd(R[i+1], A[i])

max_gcd = 0
for i in range(N):
    if gcd(L[i], R[i+1]) > max_gcd:
        max_gcd = gcd(L[i], R[i+1])

print '{}'.format(max_gcd)
```


### Problem D - Flipping Signs
```python
N = int(input())
A = list(map(int, raw_input().split()))

ng_cnt = 0

for i in range(N):
    if A[i] < 0:
        ng_cnt += 1
        A[i] = -A[i]

if ng_cnt % 2 == 0:
    print '{}'.format( sum(A) )
else:
    print '{}'.format( sum(A) - min(A)*2)
```
